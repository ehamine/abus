Detailed Description

A-Bus method RPC.

Usage: 

        cABusRequestMethod *my_rpc = my_abus->RequestMethod("examplesvc", "sum");

        my_rpc->append_int("a", 5);
        my_rpc->append_int("b", 3);
        if (my_rpc->invoke(ABUS_RPC_FLAG_NONE, 1000) == 0) {
                int res;
                my_rpc->get_int("sum_result", &res);
        }

        delete my_rpc;
